JSON OBJ STRUCTURE:

"state": (type: string, default: "0x0000") - script id to match when loading move.
"script_type": (type: string, default: "01") - sub-type of script (Ex: [A5 {type = "01"} 02] for A5\01.json or [25 {type = "03"} 02] for 25\03.json)
"name": (type: string, default: "NO NAME") - name used to describe script purpose.
"last": (type: string, default: "") - name to show in true / false checks relating to the previous check.
"no_arg_text": (type: string, default: "") - text to show when there are 0 args.
"args": (type: list) - positional input parameters for the script.
	"index": (type: int or list[int], default: arg index) - index of args_list used to get value.
	"prefix": (type: string, default: "") - text to place before value. Ex: prefix = "MOVE", name = "id" => MOVE[id:##] 
	"name": (type: string, default: arg index) - name of parameter.
	"value_type": (type: string, default: null) - value set used to get name for value.
	"format_method": (type: string, default: "format_value") - function used to format value ("format_value" is used 99% of the time. Set index to -1 to use script id as first value when using: "name_from_enum".).
	"data": (type = "format_value"):
		"auto": (type: bool, default: False) - if True the value 0 is replaced with 'Auto'.
		"decode": (type: bool, default: False) - if True value will be treated as an encoded value and decoded.
		"negative": (type: bool, default: False) - if True value will be treated as a signed int and be shown as a negative number if needed.
		"prefix": (type: string, default: "") - text to put before value.
		"suffix": (type: string, default: "") - text to put after value.
		"offset": (type: int, default: 0) - number to add to value
		"replace_char": (type: string, default: " ") - This replaces underscores ( _ ) in value names.
		"format": (type: bool, default: False) - Capitalizes name of value.
		"slice": (type: bool, default: False) - Separates parts of the name of a value into an array using _ as a delimiter.
		"slice_index": (type: int, default: 0) - index of string to use from array generated by 'slice' being True.
		"encoded_percent": (type: bool, default: False) - if True value will be treated as an encoded percent, using 'percent_base' as the 100% value.
		"percent_base": (type: int, default: 15360 (0x3c00)) - value to be used as 100% when decoding an encoded percent.

	"data": (type = "name_from_enum"):
		"replace_char": (type: string, default: " ") - This replaces underscores ( _ ) in value names.
		"format": (type: bool, default: False) - Capitalizes name of value.
		"slice": (type: bool, default: False) - Separates parts of the name of a value into an array using _ as a delimiter.
		"slice_index": (type: int, default: 0) - index of string to use from array generated by 'slice' being True.
